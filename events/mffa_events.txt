add_namespace = mffa

country_event = {
	id = mffa.1
	
	hidden = yes
	
	fire_only_once = yes
	
	is_triggered_only = yes
	
	immediate = {
		add_threat = -60
	}
	
	option = { name = mffa.1.a }
}

country_event = {
	id = mffa.2
	
	hidden = yes
	
	fire_only_once = yes
	
	mean_time_to_happen = { days = 2 }	
	
	trigger = {
		OR = {
			original_tag = ARG
			original_tag = URG
			original_tag = PAR
			original_tag = CHL
			original_tag = BRA
			original_tag = BOL
			original_tag = PRU
			original_tag = ECU
			original_tag = COL
			original_tag = VEN
		}
		has_war = no
		has_country_flag = our_war_is_not_yours
	}
	
	immediate = {
		every_state = {
			remove_claim_by = ARG
			remove_claim_by = URG
			remove_claim_by = PAR
			remove_claim_by = CHL
			remove_claim_by = BRA
			remove_claim_by = BOL
			remove_claim_by = PRU
			remove_claim_by = ECU
			remove_claim_by = COL
			remove_claim_by = VEN
		}
		if = {
			limit = {
				country_exists = ARG
			}
			ARG = {
				add_state_claim = 299 # Фолклендские острова
				add_state_claim = 720 # Остров Южная Георгия
			}
		}
		if = {
			limit = {
				has_idea = mffa_us_influence_shield
			}
			remove_ideas = mffa_us_influence_shield
		}
	}
	
	option = { name = mffa.1.a }
}
