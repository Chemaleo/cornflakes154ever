country_event = {
	id = cf154ever_startup.8
	title = cf154ever_startup.2.t
	desc = cf154ever_startup.1.d
	picture = GFX_report_event_001
	
	fire_only_once = no
	
	is_triggered_only = yes
	
	option = {
		name = cf154ever_startup.2.b
		custom_effect_tooltip = CF154EVER_SETTINGS_TT
		if = {
			limit = {
				NOT = {
					has_global_flag = CF154EVER_ALL_EXPLOITS_REMOVED_TRUE
					has_global_flag = CF154EVER_ALL_CRUTCHES_REMOVED_TRUE
				}
			}
			hidden_effect = {
				country_event = { id = cf154ever_startup.3 }
			}
		}
		else = {
			hidden_effect = {
				country_event = { id = cf154ever_startup.9 }
			}
		}
	}
	
	option = {
		name = cf154ever_startup.2.c
		custom_effect_tooltip = CF154EVER_EXIT_TT
		if = { limit = { has_global_flag = CF154EVER_ALL_EXPLOITS_REMOVED_TRUE }
		clr_global_flag = CF154EVER_ALL_EXPLOITS_REMOVED_TRUE
		}
		if = { limit = { has_global_flag = CF154EVER_ALL_CRUTCHES_REMOVED_TRUE }
		clr_global_flag = CF154EVER_ALL_CRUTCHES_REMOVED_TRUE
		}
	}
}
